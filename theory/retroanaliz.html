Ретроанализ

Здесь будет представлен код для популярной задачи "Ретроанализ для маленьких".

#include<bits/stdc++.h>

using namespace std;

void solve(int n,int m) {
    vector<int> g[n],rg[n];
    for(int i=0;i<m;i++) {
        int x,y;
        cin>>x>>y;
        g[x-1].push_back(y-1);
        rg[y-1].push_back(x-1);
    }
    vector<int> res(n,0);
    vector<int> cnt(n);
    vector<int> moves(n,-1);
    queue<int> q;
    for(int i=0;i<n;i++) {
        cnt[i]=g[i].size();
        if (cnt[i]==0) {
            res[i]=-1;
            moves[i]=0;
            q.push(i);
        }
    }
    while(!q.empty()) {
        int v = q.front();
        q.pop();
        if (res[v]==-1) {
            for(int u : rg[v]) {
                if (res[u]==0) {
                    res[u] = 1;
                    moves[u] = moves[v] + 1;
                    q.push(u);
                }
            }
        } else {
            for(int u : rg[v]) {
                if (res[u]==0) {
                    cnt[u]--;
                    if (cnt[u]==0) {
                        res[u] = -1;
                        moves[u] = moves[v] + 1;
                        q.push(u);
                    }
                }
            }
        }
    }

    for(int i=0;i<n;i++) {
        if (res[i]==0) cout<<"DRAW"<<'\n'; else
        if (res[i]==-1) cout<<"SECOND"<<'\n'; else
        cout<<"FIRST"<<'\n';
    }
    cout<<'\n';

}


int main() {

	ios_base::sync_with_stdio(false);
	cin.tie(0);

    int n,m;
    while(cin>>n>>m) {
        solve(n,m);
    }

    return 0;
}
